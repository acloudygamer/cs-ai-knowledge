# 类与模块

## Class（类）

### 基本语法

```javascript
// 类声明（不可提升）
class Person {
  // 构造函数
  constructor(name, age) {
    this.name = name;
    this.age = age;
  }

  // 实例方法
  greet() {
    return `Hello, I am ${this.name}`;
  }

  // getter
  get info() {
    return `${this.name}, ${this.age} years old`;
  }

  // static 方法（类方法，非实例方法）
  static create(name, age) {
    return new Person(name, age);
  }
}

const p = new Person('Alice', 30);
p.greet();          // 'Hello, I am Alice'
Person.create('Bob', 25);  // static factory method
```

### 继承

```javascript
class Animal {
  constructor(name) {
    this.name = name;
  }

  speak() {
    console.log(`${this.name} makes a noise`);
  }
}

class Dog extends Animal {
  constructor(name, breed) {
    super(name);  // 调用父类构造函数
    this.breed = breed;
  }

  // 重写父类方法
  speak() {
    super.speak();  // 调用父类方法
    console.log(`${this.name} barks`);
  }
}

const dog = new Dog('Rex', 'German Shepherd');
dog.speak();
// 'Rex makes a noise'
// 'Rex barks'
```

### 私有字段

```javascript
class BankAccount {
  // 私有字段（#开头）
  #balance = 0;
  #id;

  constructor(id) {
    this.#id = id;
  }

  // 公开方法访问私有字段
  deposit(amount) {
    if (amount <= 0) throw new Error('Invalid amount');
    this.#balance += amount;
  }

  withdraw(amount) {
    if (amount > this.#balance) throw new Error('Insufficient funds');
    this.#balance -= amount;
  }

  getBalance() {
    return this.#balance;
  }
}

const account = new BankAccount('ACC001');
account.deposit(1000);
account.withdraw(300);
// account.#balance  // SyntaxError: Private field
```

### 静态字段

```javascript
class Config {
  static DEFAULT_TIMEOUT = 5000;
  static MAX_RETRIES = 3;

  constructor(apiKey) {
    this.apiKey = apiKey;
  }
}

console.log(Config.DEFAULT_TIMEOUT);  // 5000
```

## ES Modules（模块系统）

### export（导出）

```javascript
// 命名导出
export const PI = 3.14159;
export function add(a, b) {
  return a + b;
}
export class MyClass { }

// 批量导出
const CONSTANT = 100;
function helper() { }
export { CONSTANT, helper };

// 重命名导出
export { helper as utilHelper };
```

### import（导入）

```javascript
// 命名导入
import { PI, add } from './math.js';
import { PI as π } from './math.js';  // 重命名

// 默认导入
import defaultExport from './module.js';

// 命名 + 默认
import defaultExport, { named1, named2 } from './module.js';

// 整体导入
import * as utils from './utils.js';
utils.add(1, 2);
```

### export default

```javascript
// 默认导出（每个模块只能有一个）
export default function main() {
  console.log('entry point');
}

// 导入默认导出
import main from './app.js';
```

### 模块导入与执行时机

```javascript
// 静态 import（编译时解析，声明提升）
import { foo } from './foo.js';  // hoisted
import { bar } from './bar.js';

// 动态 import（运行时加载）
const module = await import('./dynamic.js');

// 动态 import 适合代码分割和按需加载
async function loadFeature() {
  if (condition) {
    const { heavyFeature } = await import('./heavy.js');
    heavyFeature.init();
  }
}
```

### CommonJS 互操作

```javascript
// CommonJS 模块（Node.js 传统）
const fs = require('fs');

// ES Module 中使用 CommonJS
// 可以直接 import CommonJS 导出（但 default export 是整个 module.exports）

// CommonJS 中使用 ES Module
// 需要使用 createRequire 或 --experimental-modules
const { createRequire } = require('module');
const require = createRequire(import.meta.url);
const pkg = require('./commonjs.cjs');
```

### 模块循环依赖

```javascript
// a.js
import { b } from './b.js';
export const a = 'a';
export function getA() { return a; }

// b.js
import { getA } from './a.js';
export const b = 'b';
export function getB() {
  // 此时 a.js 可能尚未完全初始化
  console.log(getA());  // 可能得到 undefined
  return b;
}

```

## 模块模式与最佳实践

### IIFE 模式（模块化封装）

```javascript
// 传统模块模式（ES6 之前）
const Calculator = (function() {
  let result = 0;

  return {
    add(n) { result += n; return this; },
    subtract(n) { result -= n; return this; },
    getResult() { return result; },
    reset() { result = 0; return this; }
  };
})();

Calculator.add(5).subtract(2).getResult();  // 链式调用
```

### 命名空间模式

```javascript
// 将相关功能组织到单一命名空间
const App = {
  Utils: { /* ... */ },
  Services: { /* ... */ },
  Components: { /* ... */ }
};
```

### barrel 导出模式（目录聚合）

```javascript
// utils/index.js（barrel file）
export { add } from './add.js';
export { subtract } from './subtract.js';
export { multiply } from './multiply.js';

// 使用时只需从 index 导入
import { add, multiply } from './utils/index.js';
```
