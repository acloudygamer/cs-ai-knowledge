# 并发与异步

## threading / multiprocessing

```python
import threading
import multiprocessing
from concurrent.futures import ThreadPoolExecutor, ProcessPoolExecutor

# threading（I/O密集型）
def fetch_url(url):
    import requests
    return requests.get(url).status_code

with ThreadPoolExecutor(max_workers=10) as executor:
    urls = ["http://example.com"] * 100
    results = list(executor.map(fetch_url, urls))

# 线程同步
lock = threading.Lock()
counter = 0

def increment():
    global counter
    with lock:                 # 保护临界区
        counter += 1

threads = [threading.Thread(target=increment) for _ in range(1000)]
for t in threads:
    t.start()
for t in threads:
    t.join()

print(counter)                 # 1000

# multiprocessing（CPU密集型）
def cpu_task(n):
    return sum(i*i for i in range(n))

if __name__ == "__main__":
    with ProcessPoolExecutor(max_workers=4) as executor:
        results = list(executor.map(cpu_task, [10**6]*8))

# 使用 Queue 进程间通信
from multiprocessing import Process, Queue

def producer(q):
    for i in range(5):
        q.put(i)

def consumer(q):
    while True:
        item = q.get()
        if item is None:
            break
        print(f"Got: {item}")

q = Queue()
p1 = Process(target=producer, args=(q,))
p2 = Process(target=consumer, args=(q,))
p1.start()
p2.start()
p1.join()
q.put(None)                    # 发送结束信号
p2.join()
```

## asyncio

```python
import asyncio

# 基础协程
async def fetch(url):
    import aiohttp
    async with aiohttp.ClientSession() as session:
        async with session.get(url) as response:
            return await response.json()

# 运行协程
async def main():
    result = await fetch("https://api.example.com/data")
    print(result)

asyncio.run(main())

# 并发执行多个任务
async def main():
    import aiohttp
    urls = ["http://example.com"] * 10

    async with aiohttp.ClientSession() as session:
        tasks = [fetch(url) for url in urls]
        results = await asyncio.gather(*tasks)

# 异步迭代
async def async_generator():
    for i in range(5):
        await asyncio.sleep(0.1)
        yield i

async def consume():
    async for item in async_generator():
        print(item)

# 异步上下文管理器
class AsyncResource:
    async def __aenter__(self):
        self.resource = await acquire()
        return self.resource

    async def __aexit__(self, exc_type, exc_val, exc_tb):
        await release(self.resource)

async def use_resource():
    async with AsyncResource() as resource:
        await resource.do_something()

# asyncio.TaskGroup（Python 3.11+）
async def main():
    async with asyncio.TaskGroup() as tg:
        task1 = tg.create_task(fetch("url1"))
        task2 = tg.create_task(fetch("url2"))

    results = [task1.result(), task2.result()]

## TaskGroup vs asyncio.gather

| 特性 | TaskGroup | asyncio.gather |
|------|-----------|----------------|
| 异常处理 | 取消所有任务，抛出 ExceptionGroup | 抛出第一个异常，其余任务继续 |
| 结果获取 | 通过 task.result() | 直接返回结果列表 |
| 细粒度控制 | 可单独取消某个任务 | 只能全部取消 |
| 作用域 | 替代 nest taskscope | 适合简单并发场景 |

```python
# gather：任一任务异常立即抛出，其余任务继续运行
async def main():
    try:
        results = await asyncio.gather(
            fetch("url1"),
            fetch("url2"),
            fetch("url3"),
        )
    except Exception as e:
        print(f"某个任务失败: {e}")

# TaskGroup：所有任务在同一个 scope，全部完成后才退出
# 任意任务异常，取消其余任务，汇总抛出 ExceptionGroup
async def main():
    try:
        async with asyncio.TaskGroup() as tg:
            tg.create_task(fetch("url1"))
            tg.create_task(fetch("url2"))
            tg.create_task(fetch("url3"))
    except* ExceptionGroup as eg:
        print(f"异常组: {eg}")
```

## 异常处理：ExceptionGroup

```python
# TaskGroup 捕获多个任务异常
async def faulty_task(n):
    await asyncio.sleep(0.1)
    raise ValueError(f"error in task {n}")

async def main():
    try:
        async with asyncio.TaskGroup() as tg:
            tg.create_task(faulty_task(1))
            tg.create_task(faulty_task(2))
            tg.create_task(faulty_task(3))
    except* ExceptionGroup as eg:
        print(f"捕获 {len(eg.exceptions)} 个异常:")
        for e in eg.exceptions:
            print(f"  - {type(e).__name__}: {e}")
        # 输出: 捕获 3 个异常:
        #   - ValueError: error in task 1
        #   - ValueError: error in task 2
        #   - ValueError: error in task 3

# 处理部分成功的结果（gather 更适合）
async def main():
    async def safe_task(n):
        await asyncio.sleep(0.1)
        if n == 2:
            raise ValueError("task 2 failed")
        return n * 2

    results = await asyncio.gather(
        safe_task(1),
        safe_task(2),
        safe_task(3),
        return_exceptions=True,  # 不抛出异常，返回异常对象
    )
    # results: [2, ValueError('task 2 failed'), 6]
```

## 超时处理

```python
# asyncio.wait_for：超时后取消任务
async def long_task():
    await asyncio.sleep(10)
    return "done"

async def main():
    try:
        result = await asyncio.wait_for(long_task(), timeout=3.0)
    except asyncio.TimeoutError:
        print("任务超时被取消")

# asyncio.timeout（Python 3.11+）：替代 wait_for，语法更简洁
async def main():
    try:
        async with asyncio.timeout(3.0):
            result = await long_task()
    except asyncio.TimeoutError:
        print("超时")

# asyncio.timeout(None)：禁用超时
async def main():
    try:
        async with asyncio.timeout(None) as cm:
            result = await long_task()
    except asyncio.TimeoutError:
        # 不会发生
        pass

# 取消正在运行的任务
async def cancellable_task():
    try:
        for i in range(10):
            await asyncio.sleep(1)
            print(f"step {i}")
    except asyncio.CancelledError:
        print("任务被取消")
        raise  # 重新抛出以符合规范

async def main():
    task = asyncio.create_task(cancellable_task())
    await asyncio.sleep(3)
    task.cancel()  # 发送 CancelledError
    try:
        await task
    except asyncio.CancelledError:
        print("已确认取消")
```

## create_task vs gather

```python
# gather：适合简单场景，结果顺序与输入顺序一致
async def main():
    urls = ["url1", "url2", "url3"]
    tasks = [fetch(url) for url in urls]
    results = await asyncio.gather(*tasks)
    # results[0] 对应 urls[0] 的结果

# create_task + 单独 await：适合需要细粒度控制的场景
async def main():
    task1 = asyncio.create_task(fetch("url1"))
    task2 = asyncio.create_task(fetch("url2"))

    # 可以在等待期间做其他事
    await asyncio.sleep(0.5)
    print("中途执行其他操作")

    # 按特定顺序获取结果
    result2 = await task2
    result1 = await task1

# 任务启动时机不同
async def main():
    # gather：所有任务几乎同时启动
    results = await asyncio.gather(
        asyncio.create_task(task_a()),
        asyncio.create_task(task_b()),
    )

    # 顺序 await：task_b 在 task_a 完成后才开始
    result_a = await task_a()
    result_b = await task_b()
```

## 常见陷阱

```python
# 陷阱1：忘记 await
async def main():
    task = asyncio.create_task(fetch("url"))  # 错误：任务从未执行
    await task

# 陷阱2：阻塞事件循环（禁止在 async 函数中调用阻塞操作）
async def main():
    # 错误：time.sleep 是阻塞操作
    await asyncio.sleep(1)
    # time.sleep(1)  # 不要这样做！

    # 正确：使用 asyncio.sleep
    await asyncio.sleep(1)

# 陷阱3：资源泄露（未关闭连接）
async def main():
    # 错误：session 未关闭
    session = aiohttp.ClientSession()
    await session.get("url")

    # 正确：使用 async with
    async with aiohttp.ClientSession() as session:
        await session.get("url")
    # session 自动关闭

# 陷阱4：在同步代码中调用异步函数
def sync_function():
    asyncio.run(fetch("url"))  # 正确：在同步函数中用 asyncio.run

# 陷阱5：混合使用线程和异步
async def main():
    # 错误：在 async 中使用 blocking thread
    with ThreadPoolExecutor() as executor:
        future = executor.submit(blocking_io)
        result = await asyncio.wrap_future(future)  # 需要显式转换

# 陷阱6：TaskGroup 中抛出异常后继续创建任务
async def main():
    try:
        async with asyncio.TaskGroup() as tg:
            tg.create_task(fetch("url1"))
            raise ValueError("提前抛出")  # 这会导致所有任务被取消
    except* ExceptionGroup:
        pass
```
