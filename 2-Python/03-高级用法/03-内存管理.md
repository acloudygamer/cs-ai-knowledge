# 内存管理

## 引用计数

```python
import sys

# 引用计数
a = [1, 2, 3]        # 引用计数 +1
b = a                # 引用计数 +1
print(sys.getrefcount(a))  # 3（包含临时引用）

del b                # 引用计数 -1
del a                # 引用计数 -1，列表被回收

# 循环引用问题
class Node:
    def __init__(self, value):
        self.value = value
        self.next = None

# 创建循环引用
node1 = Node(1)
node2 = Node(2)
node1.next = node2
node2.next = node1

# 需要 gc.collect() 才能回收
```

## 垃圾回收

```python
import gc

# 手动触发垃圾回收
gc.collect()

# 禁用自动回收
gc.disable()
gc.enable()

# 查看回收对象
print(gc.get_stats())

# 调试
gc.set_debug(gc.DEBUG_STATS)
gc.get_referrers(obj)      # 找出谁引用了对象

```

## Python 3.11 垃圾回收改进

Python 3.11 的 Faster CPython 项目对垃圾回收进行了多项优化：

### 零成本异常

Python 3.11 重写了异常处理，通过"零成本异常"机制提升性能。异常不再捕获时产生额外开销，只有真正抛出异常时才付出代价。

```python
# Python 3.10 及之前：try 块有轻微性能开销
def old_style():
    try:
        x = data.get('key')
        if x is None:
            raise KeyError('key')
    except KeyError:
        return 'default'

# Python 3.11：try 块几乎零开销
def new_style():
    try:
        return data['key']
    except KeyError:
        return 'default'
```

### 垃圾回收器优化

Python 3.11 对分代垃圾回收器进行了优化：

```python
import gc

# Python 3.11 新增：获取回收器配置
print(gc.get_threshold())  # (700, 10, 10) - 三代阈值

# Python 3.11 新增：设置分代回收参数
gc.set_threshold(500, 10, 10)  # 调整触发频率

# 查看各代对象数量
print(gc.get_count())  # 各代自上次回收后的对象增量

# 强制在特定代回收
gc.collect(generation=0)  # 只回收年轻一代（最频繁）
gc.collect(generation=1)  # 回收中一代
gc.collect(generation=2)  # 回收所有代
```

### 循环引用检测

循环引用是垃圾回收的难点，Python 通过标记-清除算法处理：

```python
import gc
import sys

# 创建循环引用
class Node:
    __slots__ = ('value', 'next')  # 减少每个对象的内存开销
    def __init__(self, value):
        self.value = value
        self.next = None

node1 = Node(1)
node2 = Node(2)
node1.next = node2
node2.next = node1  # 循环引用

# 在 Python 3.10+，可以通过以下方式观察回收
del node1
del node2
gc.collect()  # 手动触发回收

# 调试循环引用泄漏
gc.set_debug(gc.DEBUG_LEAK)  # 追踪泄漏对象
gc.collect()
print(gc.garbage)  # 查看无法回收的对象
```

### gc 模块常用函数速查

```python
import gc

# 基础操作
gc.collect()              # 手动触发完整垃圾回收
gc.enable()                # 启用自动回收（默认）
gc.disable()               # 禁用自动回收

# 查询操作
gc.get_count()            # 返回每代自上次回收后的对象计数 (count0, count1, count2)
gc.get_threshold()        # 返回触发各代回收的阈值 (threshold0, threshold1, threshold2)
gc.get_stats()            # 返回回收统计信息字典列表
gc.get_referrers(obj)     # 返回直接引用 obj 的对象（调试用）
gc.get_referents(*objs)   # 返回 objs 直接引用的对象

# 调试标志
gc.set_debug(flags)        # 设置调试标志
gc.DEBUG_STATS            # 打印回收统计
gc.DEBUG_LEAK              # 追踪无法回收的对象
gc.DEBUG_OBJECTS           # 追踪回收的对象
gc.DEBUG_SAVEALL           # 保存所有 unreachable 对象到 gc.garbage

# 手动管理对象（高级）
gc.freeze()                # 冻结所有未回收对象，不再扫描（Python 3.11+）
gc.unfreeze()              # 解冻，开始扫描
gc.get_freeze_count()      # 获取冻结对象数量

# 追踪特定对象
gc.callbacks.append(my_callback)  # 注册回收前后回调
# 回调函数签名: callback(phase, info)
# phase: 'start' 或 'stop'
# info: 包含统计信息的字典
```

### Python 3.11 内存优化：__slots__

`__slots__` 是 Python 3.11 优化内存的重要手段：

```python
import sys

# 普通类：每个实例有 __dict__，占用大量内存
class Point:
    def __init__(self, x, y):
        self.x = x
        self.y = y

p = Point(1, 2)
print(sys.getsizeof(p))              # 约 48 bytes（不含 __dict__ 内容）
print(sys.getsizeof(p.__dict__))      # 约 104 bytes（实例属性字典）

# 使用 __slots__：无 __dict__，内存占用大幅减少
class PointSlotted:
    __slots__ = ('x', 'y')

    def __init__(self, x, y):
        self.x = x
        self.y = y

p2 = PointSlotted(1, 2)
print(sys.getsizeof(p2))              # 约 40 bytes（更小）

# 注意：__slots__ 的对象不能动态添加属性
# p2.z = 3  # AttributeError

# 多继承时，所有父类都需要有 __slots__
class NamedPoint(PointSlotted):
    __slots__ = ('name',)  # 必须包含自己的 slot 并继续父类的 slots

np = NamedPoint(1, 2, 'origin')
print(np.name)  # 'origin'
```

### 弱引用

弱引用允许引用对象而不增加其引用计数，对缓存场景很有用：

```python
import weakref

class Cache:
    def __init__(self):
        self._cache = {}

    def get(self, key):
        ref = self._cache.get(key)
        return ref() if ref else None

    def set(self, key, value):
        self._cache[key] = weakref.ref(value)

# 对象在无强引用时可以被回收
import gc
obj = {'data': [1, 2, 3]}
ref = weakref.ref(obj)
print(ref() is obj)  # True
del obj
gc.collect()
print(ref() is None)  # True，对象已被回收
```

## 生成器

```python
# 生成器函数
def countdown(n):
    while n > 0:
        yield n
        n -= 1

for num in countdown(5):
    print(num)

# 生成器表达式
gen = (x**2 for x in range(10**6))  # 惰性，不占内存
print(next(gen))                    # 0
print(next(gen))                    # 1

# itertools 生成器
import itertools

# 无限迭代器
counter = itertools.count(1)
next(counter)                       # 1
next(counter)                       # 2

# 有限迭代器
cycle = itertools.cycle([1, 2, 3])  # 无限循环
repeat = itertools.repeat(10, 3)    # 重复3次

# 组合
pairs = itertools.product([1, 2], ['a', 'b'])
combinations = itertools.combinations([1, 2, 3], 2)
permutations = itertools.permutations([1, 2, 3], 2)

# yield from（委托生成）
def flattern(nested):
    for item in nested:
        if isinstance(item, (list, tuple)):
            yield from flattern(item)
        else:
            yield item
```
