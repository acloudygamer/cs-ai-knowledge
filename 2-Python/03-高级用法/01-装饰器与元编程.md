# 装饰器与元编程

## 装饰器模式

```python
import functools
import time

# 日志装饰器
def log_calls(func):
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        print(f"Calling {func.__name__}")
        try:
            result = func(*args, **kwargs)
            print(f"{func.__name__} returned {result}")
            return result
        except Exception as e:
            print(f"{func.__name__} raised {type(e).__name__}: {e}")
            raise
    return wrapper

# 计时装饰器
def timer(func):
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        start = time.perf_counter()
        result = func(*args, **kwargs)
        elapsed = time.perf_counter() - start
        print(f"{func.__name__} took {elapsed:.4f}s")
        return result
    return wrapper

# 缓存装饰器（memoization）
def cache(func):
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        # 构建缓存键
        key = str(args) + str(kwargs)
        if key not in wrapper._cache:
            wrapper._cache[key] = func(*args, **kwargs)
        return wrapper._cache[key]
    wrapper._cache = {}
    return wrapper

@cache
def fibonacci(n):
    if n < 2:
        return n
    return fibonacci(n-1) + fibonacci(n-2)

# 重试装饰器
def retry(max_attempts=3, delay=1):
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            for attempt in range(max_attempts):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    if attempt == max_attempts - 1:
                        raise
                    print(f"Attempt {attempt+1} failed, retrying...")
                    time.sleep(delay)
        return wrapper
    return decorator
```

## functools.wraps

```python
# functools.wraps 保留原函数元信息
def docstring_decorator(func):
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        """Wrapper docstring"""
        return func(*args, **kwargs)
    return wrapper

@docstring_decorator
def my_function():
    """This is my function"""
    pass

print(my_function.__name__)   # my_function（保留原名）
print(my_function.__doc__)    # This is my function（保留原文档）
```

```python
from functools import lru_cache

@lru_cache(maxsize=128)
def expensive_computation(n):

# functools.partial - 创建部分应用的函数
from functools import partial

def power(base, exponent):
    return base ** exponent

square = partial(power, exponent=2)
cube = partial(power, exponent=3)

print(square(5))  # 25
print(cube(5))    # 125

# functools.reduce - 函数式编程
from functools import reduce

numbers = [1, 2, 3, 4, 5]
product = reduce(lambda x, y: x * y, numbers, 1)

# functools.singledispatch - 泛函数（根据第一个参数类型分发）
from functools import singledispatch

@singledispatch
def process(data):
    """默认处理"""
    return str(data)

@process.register(int)
def _(data):
    return f"Integer: {data * 2}"

@process.register(str)
def _(data):
    return f"String: {data.upper()}"

print(process(42))        # Integer: 84
print(process("hello"))   # String: HELLO
print(process([1, 2]))    # [1, 2]
```

## 类装饰器

```python
# 类装饰器（装饰类）
def add_method(cls):
    def new_method(self):
        return f"New method on {self}"
    cls.new_method = new_method
    return cls

@add_method
class MyClass:
    def existing(self):
        return "Existing method"

obj = MyClass()
print(obj.new_method())        # New method on <MyClass instance>
print(obj.existing())          # Existing method

# __call__ 使实例可调用
class CallableClass:
    def __init__(self, func):
        self.func = func

    def __call__(self, *args, **kwargs):
        return self.func(*args, **kwargs)

@CallableClass
def greet(name):
    return f"Hello, {name}!"

print(greet("Alice"))          # Hello, Alice!

# 类装饰器：自动注册方法
class RouteRegistry:
    """方法注册器装饰器"""
    _routes = {}

    def __init__(self, path):
        self.path = path

    def __call__(self, func):
        RouteRegistry._routes[self.path] = func
        return func

registry = RouteRegistry

@registry("/home")
def home():
    return "Home Page"

@registry("/about")
def about():
    return "About Page"

print(RouteRegistry._routes)
# {'/home': <function home>, '/about': <function about>}
```

## 上下文管理器

### 基础：__enter__ / __exit__

```python
# 基本上下文管理器
class DatabaseConnection:
    def __init__(self, connection_string):
        self.connection_string = connection_string
        self.connection = None

    def __enter__(self):
        """进入上下文时调用"""
        print(f"Connecting to {self.connection_string}")
        self.connection = f"Connection-{self.connection_string}"
        return self  # 返回值绑定到 as 后的变量

    def __exit__(self, exc_type, exc_val, exc_tb):
        """退出上下文时调用"""
        if exc_type is not None:
            print(f"Exception: {exc_val}")
            return False  # 不压制异常
        print(f"Closing connection")
        self.connection = None
        return True  # 压制异常

# 使用
with DatabaseConnection("localhost:5432") as conn:
    print(f"Using {conn.connection}")
# Connecting to localhost:5432
# Using Connection-localhost:5432
# Closing connection

# 异常处理在 __exit__ 中
with DatabaseConnection("localhost:5432") as conn:
    raise ValueError("Connection failed")
# Connecting to localhost:5432
# Exception: Connection failed
# 异常被重新抛出
```

### contextlib 模块

```python
from contextlib import contextmanager
from contextlib import closing
from contextlib import suppress
from contextlib import nullcontext

# @contextmanager - 用生成器创建上下文管理器
@contextmanager
def managed_resource(name):
    """替代手写 __enter__/__exit__"""
    print(f"Acquiring {name}")
    resource = f"Resource-{name}"
    try:
        yield resource  # 相当于 __enter__ 返回值
    finally:
        print(f"Releasing {name}")  # 相当于 __exit__

with managed_resource("database") as res:
    print(f"Using {res}")
# Acquiring database
# Using Resource-database
# Releasing database

# closing - 自动调用 close() 方法
from urllib.request import urlopen

with closing(urlopen("https://example.com")) as page:
    content = page.read()

# suppress - 压制指定异常
from contextlib import suppress

with suppress(FileNotFoundError, PermissionError):
    with open("secret.txt") as f:
        content = f.read()

# nullcontext - 条件性使用上下文管理器
from contextlib import nullcontext

use_connection = False
ctx = db_connection() if use_connection else nullcontext()

with ctx as conn:
    if conn:
        print(f"Using {conn}")

# redirect_stdout - 重定向标准输出
from contextlib import redirect_stdout
import io

buffer = io.StringIO()
with redirect_stdout(buffer):
    print("This goes to buffer")

print(buffer.getvalue())  # "This goes to buffer\n"

# redirect_stderr 同理
```

### contextmanager 高级用法

```python
from contextlib import contextmanager
import time

# 带超时的上下文管理器
@contextmanager
def timeout_context(seconds):
    """在指定秒数后自动中断"""
    import signal

    def handler(signum, frame):
        raise TimeoutError(f"Operation timed out after {seconds}s")

    old_handler = signal.signal(signal.SIGALRM, handler)
    signal.alarm(seconds)
    try:
        yield
    finally:
        signal.alarm(0)
        signal.signal(signal.SIGALRM, old_handler)

# 锁的上下文管理器
import threading

@contextmanager
def lock_context(lock):
    lock.acquire()
    try:
        yield
    finally:
        lock.release()

# 更推荐：with lock: 本身已是上下文管理器
# 但对于复杂场景很有用
lock = threading.Lock()

with lock_context(lock):
    print("Critical section")

# 嵌套上下文管理器
@contextmanager
def tag(name):
    print(f"<{name}>")
    yield
    print(f"</{name}>")

with tag("html"):
    print("Content")
    with tag("body"):
        print("Page body")
# <html>
# Content
# <body>
# Page body
# </body>
# </html>

# 资源计数上下文管理器
class ResourceCounter:
    def __init__(self):
        self.active = 0

    @contextmanager
    def acquire(self):
        self.active += 1
        try:
            yield self.active
        finally:
            self.active -= 1

    def __len__(self):
        return self.active

counter = ResourceCounter()

with counter.acquire():
    print(f"Active: {len(counter)}")  # 1
    with counter.acquire():
        print(f"Active: {len(counter)}")  # 2
    print(f"Active: {len(counter)}")  # 1
print(f"Active: {len(counter)}")  # 0
```

### 常见陷阱

```python
# 陷阱1：生成器未正确实现为上下文管理器
@contextmanager
def bad_pattern():
    print("Start")
    result = setup()  # 如果这里抛异常，cleanup 不会执行
    yield result
    cleanup()  # 可能不执行

# 正确做法：使用 try-finally
@contextmanager
def good_pattern():
    print("Start")
    try:
        result = setup()
        yield result
    finally:
        cleanup()  # 总是执行

# 陷阱2：忘记处理异常
@contextmanager
def dangerous():
    yield
    # 异常被吞掉，不会传播
    cleanup()

# 正确：检查异常类型
@contextmanager
def safe():
    try:
        yield
    except Exception as e:
        print(f"Handled: {e}")
        # 可选择重新抛出
        # raise
    finally:
        cleanup()
```

### 装饰器工厂

```python
# 带参数的类装饰器
def register(name):
    """类装饰器工厂"""
    def decorator(cls):
        cls._registered_name = name
        Registry._classes[name] = cls
        return cls
    return decorator

class Registry:
    _classes = {}

@register("user")
class User:
    pass

@register("admin")
class Admin:
    pass

print(Registry._classes)
# {'user': <class User>, 'admin': <class Admin>}

# 组合多个装饰器
def debounce(seconds):
    """防抖装饰器"""
    import time
    def decorator(func):
        last_call = [0]
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            now = time.time()
            if now - last_call[0] >= seconds:
                last_call[0] = now
                return func(*args, **kwargs)
        return wrapper
    return decorator

def rate_limit(max_calls):
    """限流装饰器"""
    import time
    from collections import deque
    def decorator(func):
        calls = deque()
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            now = time.time()
            # 清理超过时间窗口的调用
            while calls and now - calls[0] >= 1:
                calls.popleft()
            if len(calls) < max_calls:
                calls.append(now)
                return func(*args, **kwargs)
            raise RuntimeError("Rate limit exceeded")
        return wrapper
    return decorator

@debounce(1)
@rate_limit(10)
def handle_request(data):
    return f"Processed {data}"
```

```python
# 类装饰器保留原类
class CountCalls:
    """统计方法调用次数"""
    def __init__(self, func):
        self.func = func
        self.call_count = 0
        functools.update_wrapper(self, func)

    def __call__(self, *args, **kwargs):
        self.call_count += 1
        print(f"{self.func.__name__} called {self.call_count} times")
        return self.func(*args, **kwargs)

@CountCalls
def fibonacci(n):
    if n < 2:
        return n
    return fibonacci(n-1) + fibonacci(n-2)

# 带配置的方法装饰器
class Retry:
    """可配置的重试机制"""
    def __init__(self, max_attempts=3, delay=1, exceptions=(Exception,)):
        self.max_attempts = max_attempts
        self.delay = delay
        self.exceptions = exceptions

    def __call__(self, func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            for attempt in range(self.max_attempts):
                try:
                    return func(*args, **kwargs)
                except self.exceptions as e:
                    if attempt == self.max_attempts - 1:
                        raise
                    time.sleep(self.delay)
        return wrapper
```

### property 和描述符

```python
# property 装饰器
class Circle:
    def __init__(self, radius):
        self.radius = radius

    @property
    def diameter(self):
        return self.radius * 2

    @property
    def area(self):
        return 3.14 * self.radius ** 2

c = Circle(5)
print(c.diameter)  # 10
c.radius = 10
print(c.diameter)  # 20

# setter 和 deleter
class User:
    def __init__(self, name):
        self._name = name

    @property
    def name(self):
        return self._name

    @name.setter
    def name(self, value):
        if not value:
            raise ValueError("Name cannot be empty")
        self._name = value

    @name.deleter
    def name(self):
        print("Deleting name")
        del self._name

user = User("Alice")
print(user.name)  # Alice
user.name = "Bob"  # 设置新值
del user.name  # 删除

# 描述符协议
class Range:
    """描述符：限制数值范围"""
    def __init__(self, min=None, max=None):
        self.min = min
        self.max = max

    def __get__(self, obj, objtype=None):
        return getattr(obj, self.name, None)

    def __set__(self, obj, value):
        if value < self.min or value > self.max:
            raise ValueError(f"Value must be between {self.min} and {self.max}")
        obj.__dict__[self.name] = value

    def __set_name__(self, obj, name):
        self.name = name

class Point:
    x = Range(0, 100)
    y = Range(0, 100)

    def __init__(self, x, y):
        self.x = x
        self.y = y

p = Point(50, 30)
p.x = 150  # ValueError: Value must be between 0 and 100
```
