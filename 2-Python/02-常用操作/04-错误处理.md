# 错误处理

## try/except/finally

```python
# 基本结构
try:
    result = 10 / 0
except ZeroDivisionError as e:
    print(f"不能除零: {e}")
except Exception as e:
    print(f"其他错误: {e}")
finally:
    print("总是执行")

# 捕获多个异常
try:
    int("hello")
except (ValueError, TypeError) as e:
    print(f"转换错误: {e}")

# 获取异常信息
try:
    raise ValueError("无效的值")
except ValueError as e:
    print(f"错误信息: {e}")
    print(f"异常类型: {type(e).__name__}")
    import traceback
    traceback.print_exc()
```

## 异常链

Python 3 支持异常链，可以在重新抛出异常时保留原始异常的上下文。

```python
# 显式异常链（raise ... from）
try:
    int("not_a_number")
except ValueError as e:
    raise RuntimeError("转换失败") from e

# 输出：
# Traceback (most recent call last):
#   File "<stdin>", line 2, in <module>
# ValueError: invalid literal for int() with base 10: 'not_a_number'
#
# The above exception was the direct cause of the following exception:
# Traceback (most recent call last):
#   File "<stdin>", line 4, in <module>
# RuntimeError: 转换失败

# 获取原始异常
try:
    int("not_a_number")
except ValueError as e:
    try:
        raise RuntimeError("转换失败") from e
    except RuntimeError as re:
        print(f"原始异常: {re.__cause__}")  # ValueError: ...
        print(f"原始异常类型: {type(re.__cause__).__name__}")  # ValueError

# 隐式异常链（Python 3 自动添加）
# 在异常处理中抛出新异常时，Python 自动设置 __context__
try:
    int("not_a_number")
except ValueError:
    raise RuntimeError("转换失败")  # __suppress_context__ = False

# 压制异常链
try:
    int("not_a_number")
except ValueError:
    raise RuntimeError("转换失败") from None  # 不显示原始异常
```

## 自定义异常

```python
# 自定义异常基类
class AppError(Exception):
    """应用异常基类"""
    def __init__(self, message, code=None):
        self.message = message
        self.code = code
        super().__init__(message)

# 具体异常类
class ValidationError(AppError):
    """验证错误"""
    def __init__(self, field, message):
        self.field = field
        super().__init__(f"{field}: {message}")

class NotFoundError(AppError):
    """资源未找到"""
    pass

class PermissionError(AppError):
    """权限错误"""
    pass

# 使用自定义异常
def validate_age(age):
    if age < 0:
        raise ValidationError("age", "年龄不能为负数")
    if age > 150:
        raise ValidationError("age", "年龄超出合理范围")
    return True

try:
    validate_age(-5)
except ValidationError as e:
    print(f"验证失败: {e.field} - {e.message}")
except AppError as e:
    print(f"应用错误: {e.message}")

# 异常转译
def load_config(path):
    try:
        with open(path) as f:
            return json.load(f)
    except FileNotFoundError as e:
        raise ConfigError(f"配置文件不存在: {path}") from e
    except json.JSONDecodeError as e:
        raise ConfigError(f"配置文件格式错误: {e.msg}") from e

class ConfigError(AppError):
    """配置错误"""
    pass
```

## 断言 assert

```python
# 断言用于开发时检查不应该发生的情况
def divide(a, b):
    assert b != 0, "除数不能为零"  # 发布时可用 -O 跳过
    return a / b

# 断言 vs 异常
# - 断言：检查"不应该发生"的错误（程序 bug）
# - 异常：检查"可能发生"的错误（用户输入、网络等）

# 常用断言模式
assert isinstance(x, int), f"Expected int, got {type(x)}"
assert 0 <= x <= 100, f"Value {x} out of range"
assert len(items) > 0, "Items cannot be empty"

# 使用 __debug__ 进行条件检查
if __debug__:
    # 仅在开发时执行
    validate_internal_state()

# -O 标志禁用所有断言
# python -O script.py
```

## warnings 模块

```python
import warnings

# 警告类别
warnings.warn("这是一个警告")  # UserWarning
warnings.warn("过期函数", DeprecationWarning)
warnings.warn("性能警告", PerformanceWarning)

# 过滤警告
warnings.filterwarnings("ignore", category=DeprecationWarning)
warnings.filterwarnings("error", category=RuntimeWarning)  # 变异常

# 警告格式
warnings.filterwarnings("always")  # 每次都显示

# 上下文管理器控制警告
with warnings.catch_warnings():
    warnings.simplefilter("ignore")
    old_function()  # 不会显示警告

# 使用装饰器抑制警告
@warnings.catch_warnings
def deprecated_function():
    warnings.warn("deprecated", DeprecationWarning)
    return "result"
```

## traceback 模块

```python
import traceback

# 打印异常堆栈
try:
    1 / 0
except Exception:
    traceback.print_exc()
# 输出：
# Traceback (most recent call last):
#   File "<stdin>", line 2, in <module>
#     1 / 0
# ZeroDivisionError: division by zero

# 获取异常字符串
try:
    1 / 0
except Exception:
    error_str = traceback.format_exc()
    print(error_str)

# 获取所有异常信息
try:
    int("not_a_number")
except Exception:
    exc_type = traceback.exception_type
    exc_value = traceback.exception_value
    exc_traceback = traceback.exception_traceback
    print(f"{exc_type.__name__}: {exc_value}")

# 手动创建堆栈
traceback.print_stack()
# 输出当前调用栈

# walk_tb - 遍历异常栈
try:
    1 / 0
except Exception:
    for filename, lineno, function, text in traceback.walk_tb(traceback.exception_traceback):
        print(f"{filename}:{lineno} in {function}")
```

## 日志记录最佳实践

```python
import logging

# 配置日志
logging.basicConfig(
    level=logging.DEBUG,
    format='%(asctime)s - %(name)s - %(levelname)s - %(message)s',
    handlers=[
        logging.FileHandler('app.log'),
        logging.StreamHandler()
    ]
)

logger = logging.getLogger(__name__)

# 异常日志记录
try:
    result = risky_operation()
except Exception as e:
    logger.exception(f"操作失败: {e}")  # 自动包含堆栈
    # 或
    logger.error(f"操作失败", exc_info=True)

# 结构化日志
logger.info("User %(user_id)s logged in", {"user_id": 123})
```

## 异常处理最佳实践

```python
# 1. 宽泛捕获 vs 精确捕获
# 宽泛捕获：捕获所有非系统退出异常
try:
    process()
except Exception as e:
    logger.error(e)

# 精确捕获：只捕获预期的异常
try:
    int(user_input)
except ValueError:
    return "Invalid input"

# 2. 异常不应该用于控制流
# 错误：频繁异常影响性能
for i in range(1000000):
    try:
        result = items[i]
    except IndexError:
        break

# 正确：使用边界检查
for i in range(min(len(items), 1000000)):
    result = items[i]

# 3. 清理资源使用 finally 或上下文管理器
# 错误
file = open("data.txt")
try:
    process(file)
finally:
    file.close()  # 如果 process 抛异常，close 可能不执行

# 正确
with open("data.txt") as file:
    process(file)  # 自动关闭

# 4. 保留异常上下文
# 错误
try:
    int("not_a_number")
except ValueError:
    raise RuntimeError("转换失败")  # 丢失上下文

# 正确
try:
    int("not_a_number")
except ValueError as e:
    raise RuntimeError("转换失败") from e

# 5. 自定义异常应该继承自合适的基类
# Exception
#   ├── RuntimeError
#   │   └── TimeoutError
#   └── ValueError
#       └── ValidationError
```
