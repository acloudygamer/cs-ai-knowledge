# 数据结构

## 列表 list

```python
# 创建
nums = [1, 2, 3]
mixed = [1, "hello", True, None]

# 添加
nums.append(4)              # 末尾添加
nums.insert(0, 0)           # 指定位置插入
nums.extend([5, 6])         # 合并列表

# 删除
nums.remove(3)              # 删除第一个匹配值
nums.pop()                  # 弹出末尾
nums.pop(0)                 # 弹出指定位置

# 查找
nums.index(2)              # 查找元素索引
nums.count(2)              # 计数

# 排序
nums.sort()                 # 原地排序
sorted(nums)                # 返回新列表
nums.reverse()              # 原地反转

# 切片
nums[1:3]                  # 切片
nums[::2]                  # 步长
nums[::-1]                 # 反转
```

## 字典 dict

```python
# 创建
person = {"name": "Alice", "age": 25}
dict([("a", 1), ("b", 2)])
dict.fromkeys(["a", "b"], 0)

# 访问
person["name"]              # 键访问（不存在报错）
person.get("gender", "unknown")  # 安全访问

# 添加/修改
person["city"] = "Beijing"

# 删除
del person["age"]
person.pop("age")

# 遍历
person.keys()               # 键
person.values()             # 值
person.items()              # 键值对

# 字典方法
person.update({"age": 26})  # 批量更新
```

## 集合 set

```python
# 创建
s = {1, 2, 3}
s = set([1, 2, 2, 3])       # 从列表创建

# 添加
s.add(4)
s.update([5, 6])

# 删除
s.remove(3)                 # 不存在会报错
s.discard(10)               # 不存在不报错
s.pop()                     # 随机删除

# 集合运算
a = {1, 2, 3}
b = {2, 3, 4}
a | b                       # 并集
a & b                       # 交集
a - b                       # 差集
a ^ b                       # 对称差集

# 子集/超集
a.issubset(b)
a.issuperset(b)
```

## 元组 tuple

```python
# 创建
point = (10, 20)
t = tuple([1, 2, 3])

# 解包
x, y = point
first, *middle, last = [1, 2, 3, 4, 5]

# 命名元组
from collections import namedtuple
Point = namedtuple('Point', ['x', 'y'])
p = Point(10, 20)
print(p.x, p.y)
```

## 其他常用数据结构

```python
# deque（双端队列）
from collections import deque
dq = deque([1, 2, 3])
dq.appendleft(0)
dq.append(4)
dq.popleft()
dq.pop()

# defaultdict（默认值字典）
from collections import defaultdict
dd = defaultdict(list)
dd[" fruits"].append("apple")

# Counter（计数器）
from collections import Counter
c = Counter(["apple", "banana", "apple"])
print(c["apple"])           # 2

# 有序字典
from collections import OrderedDict
od = OrderedDict()
od["a"] = 1
od["b"] = 2
```
