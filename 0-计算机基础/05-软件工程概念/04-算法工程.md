# 算法工程

## 概念

**算法工程**是将算法应用于实际问题的实践，关注算法在工程中的复杂度、性能和实用性。

```
问题 → 算法选择 → 实现 → 优化 → 生产部署
```

## 关系

**关键连接**：
- 算法 → **复杂度分析**：时间/空间复杂度决定算法适用场景
- 算法 → **数据结构**：不同数据结构支持不同操作
- 算法 → **实际应用**：排序/搜索/图论在工程中的具体应用
- 算法 → **面试**：算法题是技术面试的核心
- 工程 → **算法**：不是最优解最好，而是最合适的解

## 复杂度分析

### 时间复杂度
```python
# O(1) - 常数时间
def get_first_element(arr):
    return arr[0]  # 不管数组多大，取第一个元素时间不变

# O(log n) - 对数时间
def binary_search(arr, target):
    left, right = 0, len(arr) - 1
    while left <= right:
        mid = (left + right) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    return -1

# O(n) - 线性时间
def find_max(arr):
    max_val = arr[0]
    for val in arr:
        if val > max_val:
            max_val = val
    return max_val

# O(n log n) - 线性对数时间
# 排序算法如TimSort (Python sorted())

# O(n²) - 平方时间
def bubble_sort(arr):
    n = len(arr)
    for i in range(n):
        for j in range(n - i - 1):
            if arr[j] > arr[j + 1]:
                arr[j], arr[j+1] = arr[j+1], arr[j]
```

### 搜索
```python
# 生产环境中的搜索
def search_products(query, products, limit=20):
    """电商商品搜索"""
    # 1. 预处理：分词
    tokens = query.lower().split()

    # 2. 过滤：标题匹配
    candidates = [
        p for p in products
        if any(token in p['title'].lower() for token in tokens)
    ]

    # 3. 排序：相关性 + 质量分数
    def relevance_score(product):
        title_match = sum(1 for t in tokens if t in product['title'].lower())
        return title_match / len(tokens) * product['rating']

    candidates.sort(key=relevance_score, reverse=True)

    return candidates[:limit]
```

### 缓存与LRU
```python
from collections import OrderedDict

class LRUCache:
    def __init__(self, capacity: int):
        self.capacity = capacity
        self.cache = OrderedDict()

    def get(self, key: int) -> int:
        if key not in self.cache:
            return -1
        self.cache.move_to_end(key)  # 最近访问
        return self.cache[key]

    def put(self, key: int, value: int) -> None:
        if key in self.cache:
            self.cache.move_to_end(key)
        self.cache[key] = value
        if len(self.cache) > self.capacity:
            self.cache.popitem(last=False)  # 删除最旧的
```

### 图算法在工程中的应用
```python
# 社交网络好友推荐（简化版）
from collections import defaultdict

def recommend_friends(user_id, friendships):
    """基于共同好友的好友推荐"""
    # 构建邻接表
    graph = defaultdict(set)
    for u, v in friendships:
        graph[u].add(v)
        graph[v].add(u)

    # 推荐：朋友的朋友（排除已好友）
    recommendations = defaultdict(int)
    for friend in graph[user_id]:
        for suggested in graph[friend]:
            if suggested != user_id and suggested not in graph[user_id]:
                recommendations[suggested] += 1

    # 按共同好友数排序
    return sorted(recommendations.items(), key=lambda x: -x[1])
```

## 实用算法思维

### 分治法
```python
# 归并排序
def merge_sort(arr):
    if len(arr) <= 1:
        return arr

    mid = len(arr) // 2
    left = merge_sort(arr[:mid])
    right = merge_sort(arr[mid:])

    return merge(left, right)

def merge(left, right):
    result = []
    i = j = 0
    while i < len(left) and j < len(right):
        if left[i] <= right[j]:
            result.append(left[i])
            i += 1
        else:
            result.append(right[j])
            j += 1
    result.extend(left[i:])
    result.extend(right[j:])
    return result
```

### 动态规划
```python
# 爬楼梯问题
def climb_stairs(n):
    if n <= 2:
        return n
    dp = [0] * (n + 1)
    dp[1] = 1
    dp[2] = 2
    for i in range(3, n + 1):
        dp[i] = dp[i-1] + dp[i-2]
    return dp[n]

# 优化：只用两个变量
def climb_stairs_optimized(n):
    if n <= 2:
        return n
    prev, curr = 1, 2
    for _ in range(3, n + 1):
        prev, curr = curr, prev + curr
    return curr
```

### 贪心算法
```python
# 活动选择问题
def activity_selection(activities):
    """选择最多不重叠的活动"""
    # 按结束时间排序
    sorted_activities = sorted(activities, key=lambda x: x[1])

    count = 0
    last_end = 0
    for start, end in sorted_activities:
        if start >= last_end:
            count += 1
            last_end = end

    return count
```

## 字符串算法

### Trie (前缀树)
```python
class TrieNode:
    def __init__(self):
        self.children = {}
        self.is_end = False

class Trie:
    def __init__(self):
        self.root = TrieNode()

    def insert(self, word):
        node = self.root
        for ch in word:
            if ch not in node.children:
                node.children[ch] = TrieNode()
            node = node.children[ch]
        node.is_end = True

    def search(self, word):
        node = self._find_node(word)
        return node is not None and node.is_end

    def starts_with(self, prefix):
        return self._find_node(prefix) is not None

    def _find_node(self, prefix):
        node = self.root
        for ch in prefix:
            if ch not in node.children:
                return None
            node = node.children[ch]
        return node

# 应用：自动补全、IP路由表、敏感词过滤
# 时间复杂度：插入 O(m), 查询 O(m)，m为字符串长度
```

### KMP 字符串匹配
```python
def build_lps(pattern):
    """构建最长前后缀数组 (Longest Proper Prefix Suffix)"""
    lps = [0] * len(pattern)
    length = 0
    i = 1
    while i < len(pattern):
        if pattern[i] == pattern[length]:
            length += 1
            lps[i] = length
            i += 1
        else:
            if length != 0:
                length = lps[length - 1]
            else:
                lps[i] = 0
                i += 1
    return lps

def kmp_search(text, pattern):
    """KMP 匹配算法，时间 O(n+m)，空间 O(m)"""
    lps = build_lps(pattern)
    i = j = 0  # i=text索引, j=pattern索引
    results = []

    while i < len(text):
        if pattern[j] == text[i]:
            i, j = i + 1, j + 1

        if j == len(pattern):
            results.append(i - j)  # 找到匹配
            j = lps[j - 1]
        elif i < len(text) and pattern[j] != text[i]:
            if j != 0:
                j = lps[j - 1]
            else:
                i += 1

    return results

# 使用场景：文本搜索、DNA序列匹配、敏感词检测
```

## 工程中的复杂度优化

### O(n²) → O(n log n) 优化案例
```python
# 优化前：O(n²) 双重循环
def find_pairs_slow(arr, target):
    pairs = []
    for i in range(len(arr)):
        for j in range(i + 1, len(arr)):
            if arr[i] + arr[j] == target:
                pairs.append((arr[i], arr[j]))
    return pairs

# 优化后：O(n) 哈希表
def find_pairs_fast(arr, target):
    seen = set()
    pairs = []
    for num in arr:
        complement = target - num
        if complement in seen:
            pairs.append((complement, num))
        seen.add(num)
    return pairs

# 场景：两数之和，复杂度从 O(n²) 降到 O(n)
```

def external_sort(file_path, chunk_size=1000000):

    with open(file_path, 'r') as f:

### 滑动窗口
```python
# 固定窗口大小
def max_sum_fixed_window(arr, k):
    """固定大小为 k 的窗口，最大子数组和"""
    window_sum = sum(arr[:k])
    max_sum = window_sum

    for i in range(k, len(arr)):
        window_sum += arr[i] - arr[i - k]
        max_sum = max(max_sum, window_sum)

    return max_sum

# 变长窗口
def length_of_longest_substring(s):
    """无重复字符的最长子串"""
    char_index = {}
    left = 0
    max_length = 0

    for right, ch in enumerate(s):
        if ch in char_index and char_index[ch] >= left:
            left = char_index[ch] + 1
        char_index[ch] = right
        max_length = max(max_length, right - left + 1)

    return max_length

# 变长窗口 - 和至少为 target 的最短子数组
def min_subarray_len(target, nums):
    left = 0
    current_sum = 0
    result = float('inf')

    for right in range(len(nums)):
        current_sum += nums[right]
        while current_sum >= target:
            result = min(result, right - left + 1)
            current_sum -= nums[left]
            left += 1

    return result if result != float('inf') else 0
```
