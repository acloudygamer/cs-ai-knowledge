# 函数

## 多返回值（Go 特色）

```go
func divide(a, b int) (int, error) {
    if b == 0 {
        return 0, errors.New("除数不能为0")
    }
    return a / b, nil
}

result, err := divide(10, 2)
if err != nil {
    fmt.Println(err)
}
```

## 命名返回值

```go
func split(sum int) (x, y int) {
    x = sum * 4 / 9
    y = sum - x
    return  // 返回 x 和 y
}
```

## 可变参数

```go
func sum(nums ...int) int {
    total := 0
    for _, n := range nums {
        total += n
    }
    return total
}

fmt.Println(sum(1, 2, 3))  // 6
fmt.Println(sum(1, 2, 3, 4, 5))  // 15
```

## 函数作为值和闭包

```go
// 函数作为参数
func apply(fn func(int) int, x int) int {
    return fn(x)
}

result := apply(func(x int) int { return x * 2 }, 5)  // 10

// 闭包
func adder() func(int) int {
    sum := 0
    return func(x int) int {
        sum += x
        return sum
    }
}

pos := adder()
fmt.Println(pos(1))  // 1
fmt.Println(pos(2))  // 3
fmt.Println(pos(3))  // 6
```

## defer 延迟执行

```go
func readFile(name string) {
    file, err := os.Open(name)
    if err != nil {
        return
    }
    defer file.Close()  // 函数退出时执行
    // ... 读取文件
}

// defer 按栈顺序执行
func demo() {
    defer fmt.Println("1")
    defer fmt.Println("2")
    defer fmt.Println("3")
    // 输出: 3 2 1
}

// defer 常用场景
func examples() {
    // 1. 释放资源
    file, _ := os.Open("test.txt")
    defer file.Close()

    // 2. 解锁互斥锁
    mu.Lock()
    defer mu.Unlock()

    // 3. 关闭数据库连接
    db, _ := sql.Open("postgres", "...")
    defer db.Close()

    // 4. 打印函数执行时间
    start := time.Now()
    defer fmt.Println("Duration:", time.Since(start))
}

// defer 注意事项
// 1. defer 在 return 之后执行，但早于函数退出
// 2. 循环中的 defer 需要谨慎使用（累积）
// 3. 命名返回值在 defer 中可见
func namedReturnExample() (result int) {
    defer func() {
        result *= 2 // 修改命名返回值
    }()
    return 10 // result = 10，之后 defer 执行 result *= 2
}
```

## 函数式选项模式（Functional Options）

Go 中配置结构体的惯用模式，支持可选参数和默认值。

```go
type Server struct {
    addr    string
    port    int
    timeout time.Duration
    logger  *log.Logger
}

type Option func(*Server)

// 选项函数
func WithAddr(addr string) Option {
    return func(s *Server) {
        s.addr = addr
    }
}

func WithPort(port int) Option {
    return func(s *Server) {
        s.port = port
    }
}

func WithTimeout(timeout time.Duration) Option {
    return func(s *Server) {
        s.timeout = timeout
    }
}

func WithLogger(l *log.Logger) Option {
    return func(s *Server) {
        s.logger = l
    }
}

// 构造函数
func NewServer(opts ...Option) *Server {
    s := &Server{
        addr:    "localhost",
        port:    8080,
        timeout: 30 * time.Second,
        logger:  log.Default(),
    }
    for _, opt := range opts {
        opt(s)
    }
    return s
}

// 使用
server := NewServer(
    WithAddr("0.0.0.0"),
    WithPort(9090),
    WithTimeout(60*time.Second),
)
```

## 错误处理与函数设计

```go
// 好：返回错误而非 panic
func DoSomething() error {
    // ...
    return nil // 或返回具体错误
}

// 好：错误作为最后一个返回值
func Process() (Result, error) {
    return Result{}, nil
}

// 差：不要混合使用 error 和 panic
func BadExample() (int, error) {
    if something {
        panic("bad") // 不要这样做
    }
    return 0, nil
}
```
