# 函数

## 多返回值（Go 特色）

```go
func divide(a, b int) (int, error) {
    if b == 0 {
        return 0, errors.New("除数不能为0")
    }
    return a / b, nil
}

result, err := divide(10, 2)
if err != nil {
    fmt.Println(err)
}
```

## 命名返回值

```go
func split(sum int) (x, y int) {
    x = sum * 4 / 9
    y = sum - x
    return  // 返回 x 和 y
}
```

## 可变参数

```go
func sum(nums ...int) int {
    total := 0
    for _, n := range nums {
        total += n
    }
    return total
}

fmt.Println(sum(1, 2, 3))  // 6
fmt.Println(sum(1, 2, 3, 4, 5))  // 15
```

## 函数作为值和闭包

```go
// 函数作为参数
func apply(fn func(int) int, x int) int {
    return fn(x)
}

result := apply(func(x int) int { return x * 2 }, 5)  // 10

// 闭包
func adder() func(int) int {
    sum := 0
    return func(x int) int {
        sum += x
        return sum
    }
}

pos := adder()
fmt.Println(pos(1))  // 1
fmt.Println(pos(2))  // 3
fmt.Println(pos(3))  // 6
```

## defer 延迟执行

```go
func readFile(name string) {
    file, err := os.Open(name)
    if err != nil {
        return
    }
    defer file.Close()  // 函数退出时执行
    // ... 读取文件
}

// defer 按栈顺序执行
func demo() {
    defer fmt.Println("1")
    defer fmt.Println("2")
    defer fmt.Println("3")
    // 输出: 3 2 1
}
```
