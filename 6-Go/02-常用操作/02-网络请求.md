# 网络请求

## net/http（标准库）

```go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io"
    "net/http"
)

func main() {
    // GET 请求
    resp, err := http.Get("https://httpbin.org/get")
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()

    body, _ := io.ReadAll(resp.Body)
    fmt.Println(string(body))

    // POST 请求
    data := map[string]string{"name": "go", "version": "1.21"}
    jsonData, _ := json.Marshal(data)

    req, _ := http.NewRequest("POST", "https://httpbin.org/post",
        io.NopCloser(bytes.NewBuffer(jsonData)))
    req.Header.Set("Content-Type", "application/json")

    client := &http.Client{}
    resp, err = client.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()

    body, _ = io.ReadAll(resp.Body)
    fmt.Println(string(body))
}
```

## 第三方库（fasthttp / gin）

### Gin 框架详解

```go
// Gin 框架（最流行）
package main

import (
    "github.com/gin-gonic/gin"
)

func main() {
    r := gin.Default()

    r.GET("/ping", func(c *gin.Context) {
        c.JSON(200, gin.H{"message": "pong"})
    })

    r.GET("/hello/:name", func(c *gin.Context) {
        name := c.Param("name")
        c.JSON(200, gin.H{"message": "Hello " + name})
    })

    r.POST("/data", func(c *gin.Context) {
        var data map[string]interface{}
        if err := c.ShouldBindJSON(&data); err != nil {
            c.JSON(400, gin.H{"error": err.Error()})
            return
        }
        c.JSON(200, data)
    })

    r.Run(":8080")
}
```

### 路由组与中间件

```go
func main() {
    r := gin.Default()

    // 全局中间件
    r.Use(gin.Logger())
    r.Use(gin.Recovery())

    // 路由组 v1
    v1 := r.Group("/api/v1")
    v1.Use(AuthMiddleware())
    {
        v1.GET("/users", listUsers)
        v1.POST("/users", createUser)
        v1.GET("/users/:id", getUser)
        v1.PUT("/users/:id", updateUser)
        v1.DELETE("/users/:id", deleteUser)
    }

    // 路由组 v2
    v2 := r.Group("/api/v2")
    {
        v2.GET("/posts", listPosts)
        v2.POST("/posts", createPost)
    }

    r.Run(":8080")
}

// 中间件示例
func AuthMiddleware() gin.HandlerFunc {
    return func(c *gin.Context) {
        token := c.GetHeader("Authorization")
        if token == "" {
            c.JSON(401, gin.H{"error": "unauthorized"})
            c.Abort()
            return
        }

        // 验证 token...
        userID := validateToken(token)
        if userID == "" {
            c.JSON(401, gin.H{"error": "invalid token"})
            c.Abort()
            return
        }

        c.Set("userID", userID)
        c.Next()
    }
}
```

### 请求绑定与验证

```go
type UserInput struct {
    Name  string `json:"name" binding:"required,min=2,max=100"`
    Email string `json:"email" binding:"required,email"`
    Age   int    `json:"age" binding:"gte=0,lte=150"`
}

func createUser(c *gin.Context) {
    var input UserInput
    if err := c.ShouldBindJSON(&input); err != nil {
        c.JSON(400, gin.H{
            "error":   "validation failed",
            "details": err.Error(),
        })
        return
    }

    // 使用 input...
    c.JSON(201, gin.H{"message": "user created"})
}

// 多种绑定方式
func bindExamples(c *gin.Context) {
    // JSON
    var jsonData map[string]interface{}
    c.ShouldBindJSON(&jsonData)

    // Form
    var formData struct {
        Name string `form:"name"`
    }
    c.ShouldBind(&formData)

    // Query
    var query struct {
        Page  int    `form:"page,default=1"`
        Limit int    `form:"limit,default=10"`
        Sort  string `form:"sort"`
    }
    c.ShouldBindQuery(&query)

    // Path
    var path struct {
        ID string `uri:"id"`
    }
    c.ShouldBindUri(&path)
}
```

    // 文件下载

    // 重定向
    c.Redirect(302, "/new-location")

    // 错误响应

    c.JSON(500, gin.H{"error": "internal server error"})
}

### Gin 错误处理与中间件模式

```go
// 统一错误处理中间件
func ErrorHandler() gin.HandlerFunc {
    return func(c *gin.Context) {
        c.Next()

        // 处理所有中间件和 handler 的错误
        if len(c.Errors) > 0 {
            err := c.Errors.Last()
            c.JSON(c.Writer.Status(), gin.H{
                "error": err.Error(),
            })
        }
    }
}

// 使用 panic 恢复的中间件
func RecoveryWithLog() gin.HandlerFunc {
    return func(c *gin.Context) {
        defer func() {
            if r := recover(); r != nil {
                log.Printf("Panic recovered: %v", r)
                c.JSON(500, gin.H{"error": "internal server error"})
                c.Abort()
            }
        }()
        c.Next()
    }
}
```

### fasthttp（高性能场景）

```go
import "github.com/valyala/fasthttp"

func fastHTTPHandler(ctx *fasthttp.RequestCtx) {
    switch string(ctx.Path()) {
    case "/ping":
        ctx.JSON(200, gin.H{"message": "pong"})
    case "/hello":
        name := ctx.UserValue("name").(string)
        ctx.JSON(200, gin.H{"message": "Hello " + name})
    default:
        ctx.Error("not found", 404)
    }
}

func main() {
    fasthttp.ListenAndServe(":8080", fastHTTPHandler)
}
```
