# 并查集 (Union-Find)

## 核心概念

并查集是一种处理不相交集合合并与查询的数据结构，主要支持两种操作：
- **Find**：查找元素所属集合的根节点
- **Union**：合并两个不相交的集合

常用于连通性判断、朋友圈问题、岛屿数量等。

## 基本实现

```python
class UnionFind:
    def __init__(self, n):
        self.parent = list(range(n))  # 父节点
        self.rank = [0] * n            # 树的秩（用于优化）

    def find(self, x):
        """路径压缩"""
        if self.parent[x] != x:
            self.parent[x] = self.find(self.parent[x])  # 递归压缩路径
        return self.parent[x]

    def union(self, x, y):
        """按秩合并"""
        root_x, root_y = self.find(x), self.find(y)
        if root_x == root_y:
            return False  # 已在同一集合

        # 小秩合并到大秩
        if self.rank[root_x] < self.rank[root_y]:
            root_x, root_y = root_y, root_x
        self.parent[root_y] = root_x
        if self.rank[root_x] == self.rank[root_y]:
            self.rank[root_x] += 1
        return True
```

## 复杂度分析

| 操作 | 朴素版 | 路径压缩 | 路径压缩 + 按秩合并 |
|------|--------|----------|-------------------|
| Find | O(n) | O(α(n)) | O(α(n)) |
| Union | O(n) | O(α(n)) | O(α(n)) |

其中 α(n) 是 Ackermann 函数的反函数，近似为常数。

## 经典问题

### 省份数量（连通分量）

```python
def find_circle_num(is_connected):
    n = len(is_connected)
    uf = UnionFind(n)
    for i in range(n):
        for j in range(i + 1, n):
            if is_connected[i][j]:
                uf.union(i, j)
    return sum(1 for i in range(n) if uf.find(i) == i)

# 更简洁的BFS版本
from collections import deque
def find_circle_num_bfs(is_connected):
    n = len(is_connected)
    visited = set()
    provinces = 0
    for i in range(n):
        if i not in visited:
            provinces += 1
            queue = deque([i])
            while queue:
                j = queue.popleft()
                if j in visited:
                    continue
                visited.add(j)
                for k in range(n):
                    if is_connected[j][k] and k not in visited:
                        queue.append(k)
    return provinces
```

### 岛屿数量（感染扩散）

```python
def num_islands(grid):
    if not grid:
        return 0
    m, n = len(grid), len(grid[0])
    uf = UnionFind(m * n)
    directions = [(1, 0), (0, 1)]

    # 只在横向和纵向合并
    for i in range(m):
        for j in range(n):
            if grid[i][j] == '1':
                idx = i * n + j
                for di, dj in directions:
                    ni, nj = i + di, j + dj
                    if 0 <= ni < m and 0 <= nj < n and grid[ni][nj] == '1':
                        uf.union(idx, ni * n + nj)

    # 统计根节点数量
    islands = set()
    for i in range(m):
        for j in range(n):
            if grid[i][j] == '1':
                islands.add(uf.find(i * n + j))
    return len(islands)
```

### 合并区间（区间集合）

```python
def merge_intervals(intervals):
    if not intervals:
        return []
    # 按起点排序，并查集连接重叠区间
    intervals.sort()
    uf = UnionFind(len(intervals))
    for i in range(len(intervals) - 1):
        if intervals[i][1] >= intervals[i + 1][0]:
            uf.union(i, i + 1)

    # 合并同一集合的区间
    merged = {}
    for i in range(len(intervals)):
        root = uf.find(i)
        if root not in merged:
            merged[root] = intervals[i][:]
        else:
            merged[root][1] = max(merged[root][1], intervals[i][1])
    return list(merged.values())
```

### 等式方程可满足性

```python
def equations_possible(equations):
    uf = UnionFind(26)
    # 第一遍：处理相等方程
    for eq in equations:
        if eq[1] == '=':
            uf.union(ord(eq[0]) - 97, ord(eq[3]) - 97)

    # 第二遍：检查不等方程
    for eq in equations:
        if eq[1] == '!':
            if uf.find(ord(eq[0]) - 97) == uf.find(ord(eq[3]) - 97):
                return False
    return True
```

## 并查集 vs DFS

| 场景 | 并查集 | DFS/BFS |
|------|--------|---------|
| 动态连通性 | 更适合 | 需要重建访问状态 |
| 频繁查询两点是否连通 | O(1) | O(V+E) |
| 需要获知所有连通分量成员 | 需额外存储 | 直接可得 |
| 图结构固定不变 | 并查集略优 | 等效 |

## 进阶应用

### Kruskal 最小生成树

```python
def kruskal(n, edges):
    """edges: [(u, v, weight)]"""
    edges.sort(key=lambda x: x[2])
    uf = UnionFind(n)
    mst_weight = 0
    mst_edges = []

    for u, v, w in edges:
        if uf.union(u, v):
            mst_weight += w
            mst_edges.append((u, v, w))
            if len(mst_edges) == n - 1:
                break
    return mst_weight, mst_edges
```

### 分离集合（种类并查集）

```python
class UnionFindParity:
    """支持种类关系的并查集（食物链问题）"""
    def __init__(self, n):
        self.parent = list(range(2 * n))  # 0~n-1:同类, n~2n-1:捕食关系
        self.rank = [0] * (2 * n)

    def find(self, x):
        if self.parent[x] != x:
            self.parent[x] = self.find(self.parent[x])
        return self.parent[x]

    def union(self, x, y, d):
        # d=1表示同类，d=0表示x吃y
        fx = self.find(x)
        fy = self.find(y)
        fx_y = self.find(x + n)
        fy_y = self.find(y + n)

        if fx == fy:
            return
        self.parent[fy] = fx if d else fx_y
        self.parent[fx_y] = fy if d else fy_y
