# 动态规划

## 核心思想

- 将问题分解为子问题
- 子问题重叠，保存计算结果
- 自底向上或自顶向下求解
- 无后效性：当前状态只与之前状态有关

## DP框架

1. **定义状态** - dp[i] 表示什么？
2. **推导状态转移方程** - dp[i] = f(dp[j], ...)
3. **确定初始值** - dp[0], dp[1] 等
4. **确定遍历顺序** - 升序/降序/按依赖顺序

## 空间优化技巧

```python
# 降维优化：将 O(n^2) 空间降至 O(n) 或 O(1)
# 原则：只保留需要的先前状态

# 例：一维DP通常可只保留前1-2个状态
def optimize_1d():
    prev2, prev1 = 0, 1  # 保留前两个状态
    for i in range(2, n):
        curr = prev1 + prev2
        prev2, prev1 = prev1, curr
    return prev1

# 例：背包问题逆序遍历实现一维
def knapsack_1d_optimized():
    dp = [0] * (capacity + 1)
    for i in range(n):
        for j in range(capacity, weights[i] - 1, -1):  # 逆序！
            dp[j] = max(dp[j], dp[j - weights[i]] + values[i])
    return dp[capacity]
```

### 爬楼梯

```python
# 一维DP
def climb_stairs(n):
    if n <= 2:
        return n
    dp = [0] * (n + 1)
    dp[1], dp[2] = 1, 2
    for i in range(3, n + 1):
        dp[i] = dp[i - 1] + dp[i - 2]
    return dp[n]

# 空间优化
def climb_stairs_optimized(n):
    if n <= 2:
        return n
    prev, curr = 1, 2
    for _ in range(3, n + 1):
        prev, curr = curr, prev + curr
    return curr
```

def rob(nums):
    if not nums:
        return 0
    if len(nums) == 1:
        return nums[0]
    dp = [0] * len(nums)
    dp[0] = nums[0]
    dp[1] = max(nums[0], nums[1])

        dp[i] = max(dp[i - 1], dp[i - 2] + nums[i])

### 最长公共子序列

```python
def lcs(text1, text2):
    m, n = len(text1), len(text2)
    dp = [[0] * (n + 1) for _ in range(m + 1)]
    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if text1[i - 1] == text2[j - 1]:
                dp[i][j] = dp[i - 1][j - 1] + 1
            else:
                dp[i][j] = max(dp[i - 1][j], dp[i][j - 1])
    return dp[m][n]
```

### 编辑距离

```python
def min_distance(word1, word2):
    m, n = len(word1), len(word2)
    dp = [[0] * (n + 1) for _ in range(m + 1)]
    for i in range(m + 1):
        dp[i][0] = i
    for j in range(n + 1):
        dp[0][j] = j
    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if word1[i - 1] == word2[j - 1]:
                dp[i][j] = dp[i - 1][j - 1]
            else:
                dp[i][j] = min(dp[i - 1][j], dp[i][j - 1], dp[i - 1][j - 1]) + 1
    return dp[m][n]
```

### 背包问题

```python
# 0-1背包（空间优化）
def knapsack_01(weights, values, capacity):
    n = len(weights)
    dp = [0] * (capacity + 1)
    for i in range(n):
        for j in range(capacity, weights[i] - 1, -1):
            dp[j] = max(dp[j], dp[j - weights[i]] + values[i])
    return dp[capacity]

# 完全背包
def complete_knapsack(weights, values, capacity):
    dp = [0] * (capacity + 1)
    for i in range(len(weights)):
        for j in range(weights[i], capacity + 1):
            dp[j] = max(dp[j], dp[j - weights[i]] + values[i])
    return dp[capacity]
```

### 股票买卖最佳时机

```python
# 通用框架（最多k次交易）
# 状态编码：j=1持有 → j=2不持有 → j=3持有 → j=4不持有 → ...（奇数=持有，偶数=不持有）
def maxProfit(k, prices):
    if not prices or k == 0:
        return 0
    n = len(prices)
    if k >= n // 2:
        profit = 0
        for i in range(1, n):
            profit += max(0, prices[i] - prices[i - 1])
        return profit

    profit = [[0] * (2 * k + 1) for _ in range(n)]
    for j in range(1, 2 * k + 1, 2):
        profit[0][j] = -prices[0]  # 奇数索引表示持有状态，初始化为买入

    for i in range(1, n):
        for j in range(1, 2 * k + 1):
            if j % 2 == 1:  # 持有状态：继承或买入
                profit[i][j] = max(profit[i - 1][j], profit[i - 1][j - 1] - prices[i])
            else:  # 不持有状态：继承或卖出
                profit[i][j] = max(profit[i - 1][j], profit[i - 1][j - 1] + prices[i])
    return profit[n - 1][2 * k]

### 最长递增子序列 (LIS)

```python
# 二分优化版 O(n log n)
import bisect

def length_of_lis(nums):
    tails = []  # tails[i] = 长度为i+1的递增子序列的最小尾元素
    for num in nums:
        pos = bisect.bisect_left(tails, num)
        if pos == len(tails):
            tails.append(num)
        else:
            tails[pos] = num
    return len(tails)

# 朴素版 O(n^2)
def length_of_lis_naive(nums):
    n = len(nums)
    dp = [1] * n
    for i in range(n):
        for j in range(i):
            if nums[j] < nums[i]:
                dp[i] = max(dp[i], dp[j] + 1)
    return max(dp) if dp else 0
```

### 路径问题

```python
# 不同路径 I（只能向下或向右）
def unique_paths(m, n):
    dp = [1] * n
    for _ in range(1, m):
        for j in range(1, n):
            dp[j] += dp[j - 1]
    return dp[n - 1]

# 不同路径 II（有障碍物）
def unique_paths_with_obstacles(grid):
    m, n = len(grid), len(grid[0])
    dp = [0] * n
    dp[0] = 1 if grid[0][0] == 0 else 0

    for i in range(m):
        for j in range(n):
            if grid[i][j] == 1:
                dp[j] = 0
            elif j > 0:
                dp[j] += dp[j - 1]
    return dp[n - 1]

# 最短路径和（带障碍）
def min_path_sum(grid):
    m, n = len(grid), len(grid[0])
    dp = [[0] * n for _ in range(m)]
    dp[0][0] = grid[0][0]

    for j in range(1, n):
        dp[0][j] = dp[0][j - 1] + grid[0][j]
    for i in range(1, m):
        dp[i][0] = dp[i - 1][0] + grid[i][0]

    for i in range(1, m):
        for j in range(1, n):
            dp[i][j] = min(dp[i - 1][j], dp[i][j - 1]) + grid[i][j]
    return dp[m - 1][n - 1]
```

### 字符串分割

```python
# 单词拆分（完全背包）
def word_break(s, word_dict):
    word_set = set(word_dict)
    n = len(s)
    dp = [False] * (n + 1)
    dp[0] = True

    for i in range(1, n + 1):
        for j in range(i):
            if dp[j] and s[j:i] in word_set:
                dp[i] = True
                break
    return dp[n]

# 交错字符串
def is_interleave(s1, s2, s3):
    m, n = len(s1), len(s2)
    if m + n != len(s3):
        return False

    dp = [[False] * (n + 1) for _ in range(m + 1)]
    dp[0][0] = True

    for i in range(m + 1):
        for j in range(n + 1):
            if dp[i][j]:
                if i < m and s1[i] == s3[i + j]:
                    dp[i + 1][j] = True
                if j < n and s2[j] == s3[i + j]:
                    dp[i][j + 1] = True
    return dp[m][n]
```

### 状态压缩DP

```python
# 旅行商问题（TSP）简化版
def solve_tsp(dist):
    n = len(dist)
    dp = [[float('inf')] * n for _ in range(1 << n)]
    dp[1][0] = 0  # 从城市0出发，状态1表示已访问城市0

    for mask in range(1 << n):
        for u in range(n):
            if not (mask & (1 << u)):
                continue
            # 从u出发，尝试去未访问的城市v
            for v in range(n):
                if mask & (1 << v):
                    continue
                next_mask = mask | (1 << v)
                dp[next_mask][v] = min(dp[next_mask][v], dp[mask][u] + dist[u][v])

    # 回到起点0
    full_mask = (1 << n) - 1
    return min(dp[full_mask][u] + dist[u][0] for u in range(1, n))

# 鸡蛋掉落问题（一维状态）
def super_egg_drop(K, N):
    # dp[k][m] = k个鸡蛋扔m次能确定的最多楼层数
    # dp[k][m] = dp[k-1][m-1] + dp[k][m-1] + 1
    m = 0
    dp = [0] * (K + 1)
    while dp[K] < N:
        m += 1
        for k in range(K, 0, -1):
            dp[k] = dp[k] + dp[k - 1] + 1
    return m
```

### 区间DP

```python
# 戳气球（区间DP模板）
def min_balloons(nums):
    # 在区间两端添加1方便处理
    nums = [1] + nums + [1]
    n = len(nums)
    dp = [[0] * n for _ in range(n)]

    # len表示区间长度，从2开始（只有一个气球时不需要戳）
    for length in range(2, n):
        for left in range(n - length):
            right = left + length
            for mid in range(left + 1, right):
                dp[left][right] = max(
                    dp[left][right],
                    dp[left][mid] + dp[mid][right] + nums[left] * nums[mid] * nums[right]
                )
    return dp[0][n - 1]

# 合并石子（经典区间DP）
def min_stones(stones):
    n = len(stones)
    prefix = [0] * (n + 1)
    for i in range(n):
        prefix[i + 1] = prefix[i] + stones[i]

    dp = [[0] * n for _ in range(n)]
    # len = 区间长度
    for length in range(2, n + 1):
        for left in range(n - length + 1):
            right = left + length - 1
            dp[left][right] = float('inf')
            for mid in range(left, right):
                cost = dp[left][mid] + dp[mid + 1][right] + prefix[right + 1] - prefix[left]
                dp[left][right] = min(dp[left][right], cost)
    return dp[0][n - 1]
```

### DP分类总结

| 类型 | 状态定义 | 遍历顺序 | 经典问题 |
|------|----------|----------|----------|
| 线性DP | dp[i] | 从左到右 | 爬楼梯、 LIS |
| 矩阵DP | dp[i][j] | 行优先 | 路径问题、编辑距离 |
| 背包DP | dp[i][j] | 物品/容量双循环 | 0-1背包、完全背包 |
| 区间DP | dp[l][r] | 长度递增 | 戳气球、合并石子 |
| 状态压缩DP | dp[mask] | mask递增 | TSP、鸡蛋掉落 |
| 树形DP | dp[u] | DFS后序 | 树上选点 |
