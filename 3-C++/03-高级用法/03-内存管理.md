# 内存管理

## 堆与栈

```cpp
void func() {
    int local = 5;           // 栈上，函数结束自动释放
    int* heap_var = new int(10);  // 堆上，需手动 delete
    delete heap_var;

    // 数组
    int* arr = new int[100];
    delete[] arr;

## 智能指针（C++11）

```cpp
#include <memory>

// unique_ptr - 独占所有权
std::unique_ptr<int> p1 = std::make_unique<int>(42);
std::unique_ptr<int[]> arr = std::make_unique<int[]>(10);
// 不能复制，只能移动
std::unique_ptr<int> p2 = std::move(p1);  // p1 变为空

// shared_ptr - 共享所有权，引用计数
std::shared_ptr<int> sp1 = std::make_shared<int>(100);
std::shared_ptr<int> sp2 = sp1;  // 引用计数变为 2
// 最后一个 shared_ptr 销毁时释放内存

// weak_ptr - 弱引用，不增加引用计数
std::weak_ptr<int> wp = sp1;
if (auto locked = wp.lock()) {
    // 对象仍存在，可以使用 locked
} else {
    // 对象已被销毁
}

// 避免循环引用
class Node {
public:
    std::shared_ptr<Node> next;
    std::weak_ptr<Node> prev;  // 用 weak_ptr 打破循环
};
```

### unique_ptr 深入

独占所有权的智能指针，不能复制，只能移动：

```cpp
#include <memory>

// 基本用法
auto p1 = std::make_unique<int>(42);
std::unique_ptr<int> p2 = std::make_unique<int>(100);

// 移动转移所有权
std::unique_ptr<int> p3 = std::move(p1);  // p1 变为空，p3 拥有 42

// 释放原始指针
p3.release();  // 返回原始指针，需手动 delete

// 重置
p3.reset();  // 释放内存，p3 变为空

// 自定义删除器
auto deleter = [](FILE* f) { if (f) fclose(f); };
std::unique_ptr<FILE, decltype(deleter)> file(
    fopen("test.txt", "r"), deleter);

// 数组特化
std::unique_ptr<int[]> arr(new int[10]);
arr[0] = 42;
```

### shared_ptr 深入

共享所有权的智能指针，通过引用计数管理生命周期：

```cpp
#include <memory>

auto sp1 = std::make_shared<int>(42);
std::shared_ptr<int> sp2 = sp1;  // 引用计数变为 2

// 引用计数操作
sp1.use_count();  // 2
sp1.unique();     // false（因为 sp2 也在使用）

// weak_ptr 观察
std::weak_ptr<int> wp = sp1;
wp.use_count();   // 2，但不增加计数
wp.expired();      // false，对象仍存在

// 提升为 shared_ptr
if (auto sp = wp.lock()) {
    // sp 有效，使用 sp
} else {
    // 对象已销毁
}

// 自定义删除器
auto deleter = [](FILE* f) { fclose(f); };
std::shared_ptr<FILE> file(fopen("test.txt", "w"), deleter);

// 别名构造：共享控制块但指向另一个对象
struct Outer { int value = 42; };
auto outer = std::make_shared<Outer>();
std::shared_ptr<int> alias(outer, &outer->value);  // 共享 outer 的控制块
```

### enable_shared_from_this

安全的让对象获取自身的 shared_ptr：

```cpp
#include <memory>

class Widget : public std::enable_shared_from_this<Widget> {
public:
    std::shared_ptr<Widget> getShared() {
        return shared_from_this();  // 返回 shared_ptr
    }
};

auto w = std::make_shared<Widget>();
auto copy = w->getShared();  // 安全返回 shared_ptr

// 错误用法：裸指针不能安全创建 shared_ptr
// Widget* raw = new Widget();
// auto bad = raw->getShared();  // 未定义行为！raw 未被 shared_ptr 管理
```

### 智能指针转换

```cpp
#include <memory>

// static_pointer_cast：静态类型转换
std::shared_ptr<Base> base = std::make_shared<Derived>();
std::shared_ptr<Derived> derived = std::static_pointer_cast<Derived>(base);

// dynamic_pointer_cast：运行时类型检查
std::shared_ptr<Base> base = std::make_shared<Derived>();
std::shared_ptr<Derived> derived = std::dynamic_pointer_cast<Derived>(base);
if (derived) {
    // 转换成功
}

// const_pointer_cast：移除 const
std::shared_ptr<const int> const_sp = std::make_shared<int>(42);
std::shared_ptr<int> mutable_sp = std::const_pointer_cast<int>(const_sp);

// reinterpret_pointer_cast：位模式重解释（C++17）
std::shared_ptr<int> sp = std::make_shared<int>(42);
std::shared_ptr<void> vsp = std::reinterpret_pointer_cast<void>(sp);
```

### 循环引用问题

```cpp
#include <memory>

struct Node {
    std::shared_ptr<Node> next;
    std::weak_ptr<Node> prev;  // 必须用 weak_ptr 打破循环
};

// 错误：循环引用导致内存泄漏
struct BadNode {
    std::shared_ptr<BadNode> next;
    std::shared_ptr<BadNode> prev;  // 循环引用！
};

// 正确：用 weak_ptr 打破循环
auto n1 = std::make_shared<Node>();
auto n2 = std::make_shared<Node>();
n1->next = n2;
n2->prev = n1;  // weak_ptr 不增加计数，不会循环
```

## RAII 原则

```cpp
// Resource Acquisition Is Initialization
// 资源获取即初始化
class FileHandler {
    FILE* file_;
public:
    FileHandler(const char* path) {
        file_ = fopen(path, "r");
        if (!file_) throw std::runtime_error("Cannot open file");
    }
    ~FileHandler() {
        if (file_) fclose(file_);
    }
    // 不可复制、不可移动
};
// 文件在对象析构时自动关闭
```

### RAII 实现锁管理

```cpp
#include <mutex>

class MutexLock {
    std::mutex& mutex_;
public:
    explicit MutexLock(std::mutex& m) : mutex_(m) { mutex_.lock(); }
    ~MutexLock() { mutex_.unlock(); }
    // 不可复制、不可移动
    MutexLock(const MutexLock&) = delete;
    MutexLock& operator=(const MutexLock&) = delete;
};

std::mutex mtx;
void critical_section() {
    MutexLock lock(mtx);  // 自动加锁
    // ... 临界区代码
}  // 析构自动解锁

// C++11 推荐使用 lock_guard 或 unique_lock
void critical_section2() {
    std::lock_guard<std::mutex> lock(mtx);
    // ...
}
```

### 唯一所有权资源管理

```cpp
#include <memory>

// 独占式资源管理 - 使用 unique_ptr
class DatabaseConnection {
    Connection* conn_;
public:
    explicit DatabaseConnection(const char* conn_str)
        : conn_(connect(conn_str)) {}

    ~DatabaseConnection() {
        if (conn_) disconnect(conn_);
    }

    // 禁用复制
    DatabaseConnection(const DatabaseConnection&) = delete;
    DatabaseConnection& operator=(const DatabaseConnection&) = delete;

    // 允许移动
    DatabaseConnection(DatabaseConnection&& other) noexcept
        : conn_(other.conn_) {
        other.conn_ = nullptr;
    }

    Connection* get() const { return conn_; }
};

// 使用工厂函数返回 unique_ptr
std::unique_ptr<DatabaseConnection> create_connection(const char* conn_str) {
    return std::make_unique<DatabaseConnection>(conn_str);
}
```

## 移动语义（C++11）

移动语义避免不必要的拷贝，提升性能：

```cpp
class Buffer {
    char* data_;
    size_t size_;
public:
    Buffer(size_t size) : size_(size), data_(new char[size]) {}

    // 移动构造函数
    Buffer(Buffer&& other) noexcept : data_(other.data_), size_(other.size_) {
        other.data_ = nullptr;
        other.size_ = 0;
    }

    // 移动赋值运算符
    Buffer& operator=(Buffer&& other) noexcept {
        if (this != &other) {
            delete[] data_;
            data_ = other.data_;
            size_ = other.size_;
            other.data_ = nullptr;
            other.size_ = 0;
        }
        return *this;
    }

    ~Buffer() { delete[] data_; }
};

// 使用移动语义
Buffer create_buffer() {
    Buffer buf(1000);
    return buf;  // 移动构造，无需复制
}

int main() {
    Buffer b1(100);
    Buffer b2 = std::move(b1);  // b1 变为空，b2 获得资源

    b2 = Buffer(200);  // 移动赋值，b2 释放旧资源，接收新资源
}
```

### 移动语义与容器

```cpp
#include <vector>
#include <string>

std::vector<std::string> v;

// 移动语义避免字符串拷贝
for (int i = 0; i < 1000; i++) {
    std::string s = "item_" + std::to_string(i);
    v.push_back(std::move(s));  // 移动而非拷贝
}

// 预分配避免重新分配
std::vector<std::string> v2;
v2.reserve(1000);
for (int i = 0; i < 1000; i++) {
    v2.emplace_back("item_" + std::to_string(i));  // 原位构造
}
```

### 完美转发

保持参数的原始值类别（左值/右值）：

```cpp
template<typename T>
void process(T&& arg) {
    // 保持原始值类别转发
    do_work(std::forward<T>(arg));
}

void do_work(std::string& s) { std::cout << "lvalue: " << s << std::endl; }
void do_work(std::string&& s) { std::cout << "rvalue: " << s << std::endl; }

std::string s = "hello";
process(s);        // 调用 lvalue 版本
process("temporary");  // 调用 rvalue 版本
process(std::move(s)); // 调用 rvalue 版本
```

### 移动语义与异常

移动操作应标记 `noexcept`，否则容器可能拒绝移动而选择拷贝：

```cpp
class Data {
    std::vector<int> data_;
public:
    // 移动操作必须 noexcept
    Data(Data&& other) noexcept
        : data_(std::move(other.data_)) {}

    Data& operator=(Data&& other) noexcept {
        data_ = std::move(other.data_);
        return *this;
    }
};

// std::vector 在重新分配时会检查 noexcept
// 如果移动操作不保证不抛异常，vector 会选择拷贝而非移动
